```python
async def chat_completion_stream(
    self,
    messages: List[Dict[str, str]],
    temperature: Optional[float] = None,
    max_tokens: Optional[int] = None,
    stop: Optional[List[str]] = None,
) -> AsyncGenerator[str, None]:
    """流式输出 chat completion。
    
    Yields:
        每个 chunk 的 content 字符串
        
    Raises:
        ModelClientError: API 调用失败
    """
    payload = {
        "model": self.model,
        "messages": messages,
        "temperature": temperature or self.temperature,
        "max_tokens": max_tokens or self.max_tokens,
        "stream": True,
    }
    if stop:
        payload["stop"] = stop
    
    headers = {
        "Authorization": f"Bearer {self.api_key}",
        "Content-Type": "application/json",
    }
    
    async with httpx.AsyncClient(timeout=self.timeout) as client:
        async with client.stream(
            "POST",
            f"{self.base_url}/chat/completions",
            headers=headers,
            json=payload,
        ) as response:
            if response.status_code != 200:
                error_text = await response.aread()
                raise ModelClientError(
                    f"API Error {response.status_code}: {error_text.decode('utf-8')}"
                )
            
            async for line in response.aiter_lines():
                if line.startswith("data: "):
                    line_data = line[6:]
                    if line_data.strip() == "[DONE]":
                        break
                    try:
                        data = json.loads(line_data)
                        delta = data["choices"][0].get("delta", {})
                        if "content" in delta:
                            yield delta["content"]
                    except json.JSONDecodeError:
                        logger.error(f"Error parsing stream line: {line}")
```

```python
def build_rolling_summary(
    self,
    existing_summary: str,
    new_messages: List[Dict[str, Any]],
    first_round_messages: Optional[List[Dict[str, Any]]] = None,
) -> str:
    """构建滚动摘要。
    
    策略：
    - 保留第一轮讨论的完整内容（建立上下文）
    - 后续轮次只保留关键信息
    - 控制总长度在 max_summary_tokens 内
    
    Args:
        existing_summary: 已有摘要
        new_messages: 本轮新消息
        first_round_messages: 第一轮消息（可选，用于保留完整内容）
        
    Returns:
        更新后的摘要
    """
    if not new_messages:
        return existing_summary
    
    # 提取关键信息
    new_points = []
    for msg in new_messages:
        sender = msg.get("sender_id", msg.get("sender_type", "未知"))
        content = msg.get("content", "")
        
        # 提取关键句子（前200字）
        if len(content) > 200:
            # 尝试在句号处截断
            truncated = content[:200]
            last_period = max(truncated.rfind('。'), truncated.rfind('！'), truncated.rfind('？'))
            if last_period > 100:
                content = content[:last_period + 1]
            else:
                content = truncated + "..."
        
        new_points.append(f"[{sender}]: {content}")
    
    new_summary = "\n".join(new_points)
    
    # 构建完整摘要
    parts = []
    
    # 保留第一轮完整内容
    if first_round_messages:
        first_round_text = "\n".join([
            f"[{m.get('sender_id', m.get('sender_type', '未知'))}]: {m.get('content', '')}"
            for m in first_round_messages
        ])
        parts.append(f"## 第一轮讨论（完整）\n{first_round_text}")
    
    # 添加已有摘要
    if existing_summary:
        parts.append(f"## 历史讨论摘要\n{existing_summary}")
    
    # 添加新摘要
    parts.append(f"## 最新讨论\n{new_summary}")
    
    combined = "\n\n".join(parts)
    
    # 截断到最大长度
    max_chars = self.max_summary_tokens * self.chars_per_token
    if len(combined) > max_chars:
        # 保留第一轮和最新讨论，截断历史摘要
        if first_round_messages:
            first_round_text = "\n".join([
                f"[{m.get('sender_id', m.get('sender_type', '未知'))}]: {m.get('content', '')}"
                for m in first_round_messages
            ])
            first_round_part = f"## 第一轮讨论（完整）\n{first_round_text}"
            
            remaining_chars = max_chars - len(first_round_part) - len(new_summary) - 100
            if remaining_chars > 0 and existing_summary:
                truncated_history = existing_summary[-remaining_chars:]
                combined = f"{first_round_part}\n\n## 历史讨论摘要\n{truncated_history}\n\n## 最新讨论\n{new_summary}"
            else:
                combined = f"{first_round_part}\n\n## 最新讨论\n{new_summary}"
        else:
            combined = combined[-max_chars:]
    
    return combined
```

```python
class CryptoService:
    def __init__(self, key_file: Path | None = None) -> None:
        self._key_file = key_file or _KEY_FILE
        self._encrypt_enabled = settings.encrypt_api_keys
        
        if self._encrypt_enabled:
            # 初始化加密（原有逻辑）
            if settings.encryption_key:
                key_bytes = settings.encryption_key.encode()
                try:
                    Fernet(key_bytes)
                    self._key = key_bytes
                except ValueError:
                    logger.warning("Invalid ENCRYPTION_KEY, falling back to key file")
                    self._key = self._load_or_generate_key()
            else:
                self._key = self._load_or_generate_key()
            self._fernet = Fernet(self._key)
        else:
            logger.info("API key encryption is disabled")
    
    def encrypt(self, plaintext: str) -> str:
        """加密明文。
        
        如果加密禁用，直接返回明文。
        """
        if not plaintext:
            return ""
        
        if not self._encrypt_enabled:
            return plaintext
        
        encrypted = self._fernet.encrypt(plaintext.encode())
        return encrypted.decode()
    
    def decrypt(self, ciphertext: str) -> str:
        """解密密文。
        
        如果加密禁用，直接返回原文。
        如果解密失败，尝试直接返回原文（兼容未加密数据）。
        """
        if not ciphertext:
            return ""
        
        if not self._encrypt_enabled:
            return ciphertext
        
        try:
            decrypted = self._fernet.decrypt(ciphertext.encode())
            return decrypted.decode()
        except InvalidToken:
            # 尝试直接返回原文（可能是未加密的数据）
            logger.warning("Failed to decrypt, returning as-is (might be unencrypted)")
            return ciphertext
    
    def mask_key(self, api_key: str) -> str:
        """掩码 API Key（保持不变）。"""
        if not api_key or len(api_key) < 12:
            return "***"
        return f"{api_key[:8]}***"
```

```python
import httpx
from typing import AsyncGenerator

# 全局 HTTP 客户端（连接池）
_global_client: httpx.AsyncClient | None = None

async def get_global_client() -> httpx.AsyncClient:
    """获取全局 HTTP 客户端（懒加载）。"""
    global _global_client
    if _global_client is None:
        _global_client = httpx.AsyncClient(
            timeout=120.0,
            limits=httpx.Limits(
                max_keepalive_connections=20,
                max_connections=100,
                keepalive_expiry=30,
            ),
        )
    return _global_client

async def close_global_client() -> None:
    """关闭全局 HTTP 客户端。"""
    global _global_client
    if _global_client is not None:
        await _global_client.aclose()
        _global_client = None


class ModelClient:
    def __init__(
        self,
        base_url: str,
        api_key: str,
        model: str,
        temperature: float = 0.7,
        max_tokens: int = 4096,
        max_retries: int = 1,
        timeout: float = 120.0,
    ):
        self.base_url = base_url.rstrip("/")
        self.api_key = api_key
        self.model = model
        self.temperature = temperature
        self.max_tokens = max_tokens
        self.max_retries = max_retries
        self.timeout = timeout
    
    async def chat_completion(
        self,
        messages: List[Dict[str, str]],
        temperature: Optional[float] = None,
        max_tokens: Optional[int] = None,
        stop: Optional[List[str]] = None,
    ) -> ModelResponse:
        """发送 chat completion 请求（使用连接池）。"""
        payload = {
            "model": self.model,
            "messages": messages,
            "temperature": temperature or self.temperature,
            "max_tokens": max_tokens or self.max_tokens,
        }
        if stop:
            payload["stop"] = stop
        
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json",
        }
        
        last_error = None
        
        for attempt in range(self.max_retries + 1):
            try:
                # 使用全局客户端
                client = await get_global_client()
                response = await client.post(
                    f"{self.base_url}/chat/completions",
                    headers=headers,
                    json=payload,
                )
                
                if response.status_code != 200:
                    error_text = response.text[:500]
                    raise ModelClientError(
                        f"API returned status {response.status_code}: {error_text}"
                    )
                
                data = response.json()
                
                return ModelResponse(
                    content=data["choices"][0]["message"]["content"],
                    usage=data.get("usage", {}),
                    model=data.get("model", self.model),
                    finish_reason=data["choices"][0].get("finish_reason"),
                )
                
            except Exception as e:
                last_error = e
                if attempt < self.max_retries:
                    wait_time = 2 ** attempt
                    logger.warning(
                        "LLM API call failed, retrying",
                        attempt=attempt + 1,
                        error=str(e),
                        wait_seconds=wait_time,
                    )
                    await asyncio.sleep(wait_time)
                else:
                    logger.error(
                        "LLM API call failed after retries",
                        attempts=self.max_retries + 1,
                        error=str(e),
                    )
        
        raise ModelClientError(f"Failed after {self.max_retries + 1} attempts: {last_error}")
    
    async def chat_completion_stream(
        self,
        messages: List[Dict[str, str]],
        temperature: Optional[float] = None,
        max_tokens: Optional[int] = None,
        stop: Optional[List[str]] = None,
    ) -> AsyncGenerator[str, None]:
        """流式输出（使用连接池）。"""
        payload = {
            "model": self.model,
            "messages": messages,
            "temperature": temperature or self.temperature,
            "max_tokens": max_tokens or self.max_tokens,
            "stream": True,
        }
        if stop:
            payload["stop"] = stop
        
        headers = {
            "Authorization": f"Bearer {self.api_key}",
            "Content-Type": "application/json",
        }
        
        # 流式请求需要单独的客户端（不能复用连接池）
        async with httpx.AsyncClient(timeout=self.timeout) as client:
            async with client.stream(
                "POST",
                f"{self.base_url}/chat/completions",
                headers=headers,
                json=payload,
            ) as response:
                if response.status_code != 200:
                    error_text = await response.aread()
                    raise ModelClientError(
                        f"API Error {response.status_code}: {error_text.decode('utf-8')}"
                    )
                
                async for line in response.aiter_lines():
                    if line.startswith("data: "):
                        line_data = line[6:]
                        if line_data.strip() == "[DONE]":
                            break
                        try:
                            data = json.loads(line_data)
                            delta = data["choices"][0].get("delta", {})
                            if "content" in delta:
                                yield delta["content"]
                        except json.JSONDecodeError:
                            logger.error(f"Error parsing stream line: {line}")
```

```python
class FileIngestionService:
    # ... 现有方法保持不变 ...
    
    async def ingest_local_file(
        self,
        session: AsyncSession,
        room_id: str,
        filepath: str,
    ) -> SharedSource | None:
        """直接读取本地文件（无需上传）。
        
        Args:
            session: 数据库会话
            room_id: 房间 ID
            filepath: 本地文件路径
            
        Returns:
            创建的 SharedSource，如果文件无效则返回 None
            
        Raises:
            ValueError: 路径不安全或文件不存在
        """
        from app.utils.path_validator import validate_path_safety
        
        # 安全校验
        if not validate_path_safety(filepath):
            raise ValueError(f"不安全的文件路径: {filepath}")
        
        path = Path(filepath)
        
        # 检查文件是否存在
        if not path.exists():
            raise ValueError(f"文件不存在: {filepath}")
        
        if not path.is_file():
            raise ValueError(f"不是文件: {filepath}")
        
        # 检查扩展名
        if not is_allowed_extension(path.name):
            logger.warning("Rejected file: invalid extension", filename=path.name)
            return None
        
        # 检查文件大小（10MB 限制）
        file_size = path.stat().st_size
        max_size = 10 * 1024 * 1024  # 10MB
        if file_size > max_size:
            raise ValueError(f"文件过大: {file_size / 1024 / 1024:.1f}MB (最大 10MB)")
        
        # 读取文件内容
        try:
            import aiofiles
            async with aiofiles.open(filepath, mode='r', encoding='utf-8', errors='replace') as f:
                content = await f.read()
        except Exception as e:
            logger.error("Failed to read file", filepath=filepath, error=str(e))
            raise ValueError(f"读取文件失败: {e}")
        
        # 截断到预算限制
        if len(content) > CONTENT_BUDGET_PER_FILE:
            content = content[:CONTENT_BUDGET_PER_FILE] + "\n...(内容已截断)"
        
        # 创建 SharedSource
        source = SharedSource(
            id=str(uuid.uuid4()),
            room_id=room_id,
            source_type="file",
            path=str(path.absolute()),
            content=content,
            file_count=1,
        )
        session.add(source)
        await session.flush()
        
        logger.info("Ingested local file", source_id=source.id, filepath=filepath)
        return source
```

```python
"""路径安全校验工具。"""

from pathlib import Path

# 危险路径模式
DANGEROUS_PATTERNS = [
    "..",  # 路径遍历
    "~",   # home 目录
]

# 允许的根目录（可选，根据需求配置）
ALLOWED_ROOTS = [
    # Path.home(),  # 用户 home 目录
    # Path("/tmp"),  # 临时目录
]


def validate_path_safety(filepath: str) -> bool:
    """验证路径安全性。
    
    Args:
        filepath: 文件路径
        
    Returns:
        True 如果路径安全，False 否则
    """
    try:
        path = Path(filepath).resolve()
    except Exception:
        return False
    
    # 检查危险模式
    for pattern in DANGEROUS_PATTERNS:
        if pattern in filepath:
            return False
    
    # 如果配置了允许的根目录，检查路径是否在允许范围内
    if ALLOWED_ROOTS:
        return any(
            str(path).startswith(str(root))
            for root in ALLOWED_ROOTS
        )
    
    return True
```
